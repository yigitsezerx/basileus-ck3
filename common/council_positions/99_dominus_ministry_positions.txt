minister_personnel = {
	skill = diplomacy
	
	name = {
		first_valid = {
			triggered_desc = {
				trigger = {
					dominus_is_byzantine_ruler_trigger = yes
					liege_or_court_owner = {
						OR = {
							culture = culture:roman
							culture = culture:eastern_roman
							culture = culture:western_roman
						}
					}
				}
				desc = minister_personnel_eastern_roman
			}
			triggered_desc = {
				trigger = { dominus_is_byzantine_ruler_trigger = yes }
				desc = minister_personnel_greek
			}
			desc = minister_personnel
		}
	}

	valid_position = {
		basileus_has_ministry_access = yes
	}

	tooltip = {
		first_valid = {
			triggered_desc = {
				trigger = {
					dominus_is_byzantine_ministry_trigger = yes
					liege_or_court_owner = {
						OR = {
							culture = culture:roman
							culture = culture:eastern_roman
							culture = culture:western_roman
						}
					}
				}
				desc = game_concept_dominus_minister_of_personnel_byz_desc
			}
			triggered_desc = {
				trigger = {
					OR = {
						dominus_is_byzantine_ministry_trigger = yes
						root = {
							OR = {
								has_title = title:e_byzantium
								has_title = title:e_roman_empire
							}
						}
					}
				}
				desc = minister_personnel_greek_desc
			}
			desc = basileus_minister_of_personnel_desc
		}
	}
	
	auto_fill = {
	}

	modifier = {
		name = minister_personnel_modifier
		fellow_vassal_opinion = 5
		monthly_diplomacy_lifestyle_xp_gain_mult = 0.05
		scale = council_scaled_by_liege_tier
	}

	modifier = {
		name = minister_personnel_modifier
		monthly_prestige = 1
		scale = council_scaled_monthly_income
	}

	modifier = {
		name = minister_personnel_modifier
		monthly_influence = 1
		enemy_political_scheme_success_chance_max_add = -5
		scale = council_scaled_admin_value
	}

	council_owner_modifier = {
		name = learn_on_the_job_modifier
		diplomacy = 1
		scale = chancellor_learn_on_the_job_scale
	}

	council_owner_modifier = {
		name = reliable_house_modifier
		diplomacy = 1
		scale = chancellor_reliable_house_scale
	}

	valid_character = {
		exists = root.liege_or_court_owner
		tgp_can_be_a_minister_trigger = { COURT_OWNER = root.liege_or_court_owner }
	}

	on_get_position = {
		got_council_position_effect = yes
		if = { limit = { NOT = { root.liege_or_court_owner = { OR = { has_title = title:e_byzantium has_title = title:e_roman_empire } } } } got_minister_position_effect = { MINISTER_TITLE = e_minister_of_personnel MINISTER_POSITION = minister_personnel } }
	}

	on_fired_from_position = {
		fired_from_council_position_effect = yes
	}

	on_lose_position = {
		
	}

	portrait_animation = happy_teacher

	barbershop_data = {
		position = { 918.0 172.0 }
		click_to_front = no
	}
}

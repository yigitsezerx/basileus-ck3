# Trigger to enable ministry features in script - Made into a scripted trigger for ease of changing it if needed
tgp_has_access_to_ministry_trigger = {
	OR = {
		AND = {
			has_title = title:h_china
			government_has_flag = government_is_celestial
		}
		AND = {
            has_title = title:e_byzantium
            government_has_flag = government_is_administrative
		}
	}
}

# Alias kept so the renamed council-position files keep working.
basileus_has_ministry_access = {
	tgp_has_access_to_ministry_trigger = yes
}

tgp_can_be_a_minister_trigger = {
	can_be_councillor_basics_trigger = yes
	# Has to be celestial/admin or a non-ruler
	OR = {
		is_ruler = no
		government_has_flag = government_is_celestial
		government_has_flag = government_is_administrative
	}
	# Gender is based on laws of the liege
	OR = {
		has_variable = can_be_councillor_override
		AND = {
			is_female = yes
			$COURT_OWNER$ = {
				NOR = {
					has_realm_law = male_only_law
					has_realm_law = male_preference_law
				}
			}
		}
		AND = {
			is_male = yes
			$COURT_OWNER$ = {
				NOR = {
					has_realm_law = female_only_law
					has_realm_law = female_preference_law
				}
			}
		}
	}
	# Players are only allowed if they have enough merit or is a councillor already, unless they are friends or disciples/elders
	# We do this to prevent the AI from "randomly" requesting the player to be on their council
	trigger_if = {
		limit = {
			is_ai = no
			$COURT_OWNER$ = { is_ai = yes }
			NOT = { is_councillor_of = $COURT_OWNER$ }
		}
		OR = {
			merit_level >= merit_level_expected_empire_value
			has_relation_friend = $COURT_OWNER$
			has_relation_disciple = $COURT_OWNER$
			has_relation_elder = $COURT_OWNER$
			# A hook on the court owner lets you force a ministry seat even
			# without the empire-tier merit threshold.
			has_hook = $COURT_OWNER$
		}
	}
}

tgp_can_be_grand_marshal_trigger = {
	can_be_councillor_basics_trigger = yes
	# Has to be celestial/admin or a non-ruler
	OR = {
		is_ruler = no
		government_has_flag = government_is_celestial
		government_has_flag = government_is_administrative
	}
	# Has to be able to lead armies
	can_be_combatant_based_on_gender_trigger = { ARMY_OWNER = $COURT_OWNER$ }
}

# Check if character is the Byzantine Emperor
dominus_is_basileus_trigger = {
	has_title = title:e_byzantium
}

# Trigger for the Emperor specifically - Used for position NAMES and TOOLTIPS
dominus_is_byzantine_ruler_trigger = {
	OR = {
		# Case 1: The scope is a character (The Emperor or a candidate)
		AND = {
			is_character = yes
			is_ruler = yes
			OR = {
				has_title = title:e_byzantium
				has_title = title:e_roman_empire
			}
		}
		# Case 2: The scope is a title
		AND = {
			is_landed_title = yes
			OR = {
				this = title:e_byzantium
				this = title:e_roman_empire
			}
		}
		# Case 3: The scope is a councillor serving the Emperor
		AND = {
			is_character = yes
			is_councillor = yes
			liege_or_court_owner = {
				OR = {
					has_title = title:e_byzantium
					has_title = title:e_roman_empire
				}
			}
		}
	}
}

# Legacy alias
dominus_is_byzantine_ministry_trigger = {
	dominus_is_byzantine_ruler_trigger = yes
}

# Court trigger (For observers viewing the Emperor)
dominus_is_byzantine_court_trigger = {
	OR = {
		dominus_is_byzantine_ruler_trigger = yes
		# Fallback for observers/empty slots in the UI
		AND = {
			NOT = { exists = this }
			any_player = {
				OR = {
					has_title = title:e_byzantium
					has_title = title:e_roman_empire
				}
			}
		}
	}
}

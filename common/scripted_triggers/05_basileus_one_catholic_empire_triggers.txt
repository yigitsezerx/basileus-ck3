# =====================================================================
# BASILEUS — ONE CATHOLIC EMPIRE (game-rule gated, e_byzantium exempt).
# ---------------------------------------------------------------------
# When the `basileus_one_catholic_empire` rule is ON, a Catholic ruler
# cannot CREATE an empire while another Catholic empire already exists.
# The Eastern Roman Empire (title:e_byzantium) and its holder are fully
# exempt: e_byzantium never counts toward the limit, and its holder is
# never restricted.
#
# MECHANISM: every de jure empire's `can_create` in vanilla landed_titles
# calls the scripted trigger
#   rule_title_creation_imperial_power_projection_title_creation_trigger
# (vanilla common/scripted_triggers/00_game_rule_triggers.txt:118-260).
# We override that trigger here — copying the vanilla body verbatim and
# prepending our restriction — so the check is injected into ALL empire
# creation at a single point. The 05_ prefix loads after vanilla's 00_.
#
# NOTE: the vanilla hook is character-scoped (it can't see which empire
# is being created), so the e_byzantium exemption is keyed on the actor
# HOLDING title:e_byzantium. The only uncovered edge — a non-Byzantine
# Catholic creating e_byzantium from scratch while another Catholic
# empire exists — is negligible (Byzantium starts already created).
# =====================================================================

# True if some OTHER ruler already holds a Catholic empire. The holder of
# title:e_byzantium is excluded so Byzantium never counts toward the limit.
basileus_one_catholic_empire_exists_trigger = {
	any_ruler = {
		count >= 1
		highest_held_title_tier >= tier_empire
		faith = faith:catholic
		NOT = { this = root }
		NOT = { has_title = title:e_byzantium }
	}
}

# Restriction applied to a would-be empire creator (root = character).
basileus_one_catholic_empire_restriction_trigger = {
	trigger_if = {
		limit = {
			has_game_rule = basileus_one_catholic_empire_on
			faith = faith:catholic
			NOT = { has_title = title:e_byzantium }   # e_byzantium holder exempt
		}
		custom_tooltip = {
			text = basileus_one_catholic_empire_restriction_tt
			NOT = { basileus_one_catholic_empire_exists_trigger = yes }
		}
	}
	trigger_else = { always = yes }
}

# Override of the vanilla power-projection trigger. Body below is the
# verbatim vanilla definition (00_game_rule_triggers.txt:119-259) with our
# one-line restriction prepended.
rule_title_creation_imperial_power_projection_title_creation_trigger = {
	# One Catholic Empire — restriction for de jure empire creation.
	basileus_one_catholic_empire_restriction_trigger = yes

	trigger_if = {
		limit = {
			has_game_rule = title_creation_imperial_power_projection_any_power_advantage
			rule_title_creation_imperial_power_projection_title_creation_targeting_trigger = yes
			highest_held_title_tier <= tier_kingdom
		}
		custom_tooltip = {
			text = rule_title_creation_imperial_power_projection_title_creation_trigger.tt.any_power_advantage
			any_neighboring_top_liege_realm_owner = {
				count = all
				trigger_if = {
					limit = { highest_held_title_tier >= tier_empire }
					save_temporary_scope_as = char_temp
					OR = {
						any_close_family_member = { is_spouse_of = root }
						root = { has_strong_hook = scope:char_temp }
						root.current_military_strength >= scope:char_temp.fifty_percent_more_of_current_military_strength
					}
				}
				trigger_else = { always = yes }
			}
		}
	}
	trigger_else_if = {
		limit = {
			has_game_rule = title_creation_imperial_power_projection_hard_power_only
			rule_title_creation_imperial_power_projection_title_creation_targeting_trigger = yes
			highest_held_title_tier <= tier_kingdom
		}
		custom_tooltip = {
			text = rule_title_creation_imperial_power_projection_title_creation_trigger.tt.hard_power_only
			any_neighboring_top_liege_realm_owner = {
				count = all
				trigger_if = {
					limit = { highest_held_title_tier >= tier_empire }
					save_temporary_scope_as = char_temp
					root.current_military_strength >= scope:char_temp.fifty_percent_more_of_current_military_strength
				}
				trigger_else = { always = yes }
			}
		}
	}
	trigger_else_if = {
		limit = {
			has_game_rule = title_creation_imperial_power_projection_soft_power_only
			rule_title_creation_imperial_power_projection_title_creation_targeting_trigger = yes
			highest_held_title_tier <= tier_kingdom
		}
		custom_tooltip = {
			text = rule_title_creation_imperial_power_projection_title_creation_trigger.tt.soft_power_only
			any_neighboring_top_liege_realm_owner = {
				count = all
				trigger_if = {
					limit = { highest_held_title_tier >= tier_empire }
					save_temporary_scope_as = char_temp
					OR = {
						any_close_family_member = { is_spouse_of = root }
						root = { has_strong_hook = scope:char_temp }
					}
				}
				trigger_else = { always = yes }
			}
		}
	}
	trigger_else_if = {
		limit = {
			has_game_rule = title_creation_imperial_power_projection_hard_power_only_strict
			rule_title_creation_imperial_power_projection_title_creation_targeting_trigger = yes
			highest_held_title_tier <= tier_kingdom
		}
		custom_tooltip = {
			text = rule_title_creation_imperial_power_projection_title_creation_trigger.tt.hard_power_only_strict.neighbours
			any_neighboring_top_liege_realm_owner = {
				count = all
				trigger_if = {
					limit = { highest_held_title_tier >= tier_empire }
					save_temporary_scope_as = char_temp
					root.current_military_strength >= scope:char_temp.fifty_percent_more_of_current_military_strength
				}
				trigger_else = { always = yes }
			}
		}
		custom_tooltip = {
			text = rule_title_creation_imperial_power_projection_title_creation_trigger.tt.hard_power_only_strict.de_jure_sharers
			any_sub_realm_county = {
				count = all
				empire = {
					OR = {
						is_title_created = no
						holder = {
							save_temporary_scope_as = char_temp
							root.current_military_strength >= scope:char_temp.fifty_percent_more_of_current_military_strength
						}
					}
				}
			}
		}
	}
	trigger_else_if = {
		limit = {
			has_game_rule = title_creation_imperial_power_projection_any_power_strict
			rule_title_creation_imperial_power_projection_title_creation_targeting_trigger = yes
			highest_held_title_tier <= tier_kingdom
		}
		custom_tooltip = {
			text = rule_title_creation_imperial_power_projection_title_creation_trigger.tt.any_power_strict.neighbours
			any_neighboring_top_liege_realm_owner = {
				count = all
				trigger_if = {
					limit = { highest_held_title_tier >= tier_empire }
					save_temporary_scope_as = char_temp
					OR = {
						any_close_family_member = { is_spouse_of = root }
						root = { has_strong_hook = scope:char_temp }
						root.current_military_strength >= scope:char_temp.fifty_percent_more_of_current_military_strength
					}
				}
				trigger_else = { always = yes }
			}
		}
		custom_tooltip = {
			text = rule_title_creation_imperial_power_projection_title_creation_trigger.tt.any_power_strict.de_jure_sharers
			any_sub_realm_county = {
				count = all
				empire = {
					OR = {
						is_title_created = no
						holder = {
							save_temporary_scope_as = char_temp
							OR = {
								any_close_family_member = { is_spouse_of = root }
								root = { has_strong_hook = scope:char_temp }
								root.current_military_strength >= scope:char_temp.fifty_percent_more_of_current_military_strength
							}
						}
					}
				}
			}
		}
	}
	trigger_else = { always = yes }
}

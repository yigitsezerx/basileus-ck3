### Torture/Blind/Castrate Override ###
# Adds Autocrator XP to the vanilla punishment effect
torture_blind_castrate_disfigure_opinion_effect = {
    # ERCF: Add Autocrator XP for traditional punishments
    scope:actor = {
        if = {
            limit = { 
                has_trait = autocrator_ercf 
                OR = {
                    "$VERB$" = "blinded"
                    "$VERB$" = "castrated"
                }
            }
            ercf_autocrator_traditional_mutilation_effect = yes
        }
    }

	scope:recipient = {
		add_opinion = {
			modifier = $VERB$_me
			target = scope:actor
		}
		if = {
			limit = {
				has_opinion_modifier = {
					modifier = grateful_opinion
					target = scope:actor
				}
			}
			remove_opinion = {
				modifier = grateful_opinion
				target = scope:actor
			}
		}

		if = {
			limit = {
				primary_spouse ?= { is_alive = yes }
				NOT = { primary_spouse = { this = scope:actor } }
			}
			primary_spouse = { add_to_list = family_of_tortured }
		}
		every_close_family_member = {
			limit = { this != scope:actor }
			add_to_list = family_of_tortured
		}
		if = {
			limit = {
				any_in_list = {
					list = family_of_tortured
					count = 1
					always = yes
				}
			}
			every_in_list = {
				list = family_of_tortured
				add_opinion = {
					target = scope:actor
					modifier = $VERB$_family_member
				}
			}
		}
	}
}

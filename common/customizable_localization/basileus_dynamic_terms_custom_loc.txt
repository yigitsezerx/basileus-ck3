# ============================================================================
# Culture-conditional flavor terms (eastern_roman vs Greek fallback).
# ----------------------------------------------------------------------------
# Same convention as basileus_senate_names.txt: branch on the scope character's
# top_liege culture; eastern_roman → Latinised term, default → Greek term.
# Invoked via [GetPlayer.Custom('Fn')], including inside dynamic game-concept
# names (game_concept_patriarch / game_concept_saracen).
# ============================================================================

GetEnemyDemonym = {
	type = character

	text = {
		trigger = { top_liege = { culture = culture:eastern_roman } }
		localization_key = BASILEUS_TERM_SARACEN_LATIN
	}
	text = {
		# Default — Greek
		localization_key = BASILEUS_TERM_SARACEN_GREEK
	}
}

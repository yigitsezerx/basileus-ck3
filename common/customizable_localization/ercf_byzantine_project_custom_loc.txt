# New Custom Loc for Filtering Notifications
# This logic ensures that notifications are only shown to players in the same realm as the project.
# This silences Chinese project notifications for Byzantine players and vice versa.

defined_text = {
	name = GetGreatProjectPlannedTitle
	
	# Case 1: Project is in Player's Realm (Owner or Overlord)
	text = {
		trigger = {
			scope:great_project = {
				owner = { top_liege = root }
			}
		}
		localization_key = msg_great_project_planned_title_byz
	}
	
	# Case 2: Project is NOT in Player's Realm (Hide it)
	text = {
		trigger = {
			scope:great_project = {
				NOT = { owner = { top_liege = root } }
			}
		}
		localization_key = "empty_string"
	}
}

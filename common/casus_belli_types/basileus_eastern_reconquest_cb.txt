# =====================================================================
# BASILEUS — EASTERN RECONQUEST — custom casus belli.
# ---------------------------------------------------------------------
# basileus_roman_reclamation_cb
#   (common/casus_belli_types/00_conquest.txt). The structural plumbing
#   (target_titles iteration, create/resolve_title_and_vassal_change,
#   conquest_cb_title_transfer, etc.) is what actually transfers the
#   duchy on victory + drives the map highlight + drives the war goal
#   selection UI. Earlier minimal version was missing all of that.
#
#   Available only:
#     (a) to characters involved in basileus_eastern_reconquest_struggle
#     (b) while the current phase has the flag
#         unlocks_basileus_roman_reclamation_casus_belli (Reconquest + Triumph)
#     (c) against characters NOT involved in the struggle (interlopers)
#         whose duchies sit inside custom_theodosian_ere
# =====================================================================

basileus_roman_reclamation_cb = {
	icon = roman_reclamation
	group = conquest

	combine_into_one = yes
	should_show_war_goal_subview = yes
	mutually_exclusive_titles = { always = yes }

	attacker_score_from_occupation_scale = 150
	attacker_score_from_battles_scale = 150
	defender_score_from_battles_scale = 150

	allowed_for_character = {
		# Attacker must be involved or interloper in our struggle AND the current phase
		# must have the unlock flag.
		any_character_struggle = {
			is_struggle_type = basileus_eastern_reconquest_struggle
			has_struggle_phase_parameter = unlocks_basileus_roman_reclamation_casus_belli
		}
		NOT = {
			government_has_flag = government_is_landless_adventurer
		}
		# Declaration cooldown — the imperial war machine can only mobilise a fresh reclamation every couple of years. (set in on_declaration)
		NOT = { has_character_flag = basileus_reclamation_cb_cooldown }
	}

	allowed_against_character = {
		#scope:attacker = { #Not a regular de jure cb.
			#has_de_jure_claim_on = scope:defender
		#}
		NOR = {
			scope:defender = { target_is_liege_or_above = scope:attacker }
			scope:attacker = { target_is_liege_or_above = scope:defender }
		}
		# Defender must have at least one county in their realm inside the struggle region
		scope:defender = {
			any_realm_county = {
				title_province = {
					geographical_region = custom_theodosian_ere
				}
			}
		}
	}

	target_titles = all
	target_title_tier = duchy
	target_de_jure_regions_above = yes
	ignore_effect = change_title_holder
	ai_only_against_neighbors = yes

	valid_to_start = {
		scope:target = {
			is_landless_type_title = no
			# Target duchy must include at least one county inside the
			# Theodosian region.
			any_de_jure_county = {
				title_province = {
					geographical_region = custom_theodosian_ere
				}
			}
		}
	}

	should_invalidate = {
		NOT = {
			any_in_list = {
				list = target_titles
				any_in_de_jure_hierarchy = {
					tier = tier_county
					holder = {
						target_is_same_character_or_above = scope:defender
					}
				}
			}
		}
	}

	on_invalidated_desc = msg_invalidate_war_title
	on_invalidated = {}

	cost = {
		prestige = {
			value = minor_prestige_value
			# Scaling cost: 0.25 prestige per region county the attacker's
			# realm ALREADY holds. The more of the East you've reclaimed,
			# the dearer the next duchy — throttles snowballing. (Mirrors
			# vanilla fp2_expel_interloper's per-county cost iteration.)
			add = {
				value = 0
				scope:attacker = {
					every_realm_county = {
						limit = {
							title_province = {
								geographical_region = custom_theodosian_ere
							}
						}
						add = 0.30
					}
				}
				desc = BASILEUS_RECLAMATION_COST_SCALING
			}
			multiply = common_cb_prestige_cost_multiplier
		}
	}

	on_declaration = {
		on_declared_war = yes
		# Start the 1 year declaration cooldown on the attacker.
		scope:attacker = {
			add_character_flag = {
				flag = basileus_reclamation_cb_cooldown
				months = 12
			}
		}
	}

	# -----------------------------------------------------------------
	# VICTORY — transfer the duchy + de-jure counties properly.
	# This mirrors vanilla duchy_conquest_cb on_victory step-for-step.
	# -----------------------------------------------------------------
	on_victory_desc = {
		desc = basileus_roman_reclamation_cb_victory_desc
	}

	on_victory = {
		scope:attacker = { show_pow_release_message_effect = yes }

		# Legitimacy bump for a successful imperial reclamation.
		add_legitimacy_attacker_victory_effect = yes

		# Accolade glory if the defender outranked us going in.
		scope:attacker = { accolade_attacker_war_end_glory_gain_med_effect = yes }

		create_title_and_vassal_change = {
			type = conquest
			save_scope_as = change
			add_claim_on_loss = yes
		}

		# Collect the targeted duchy title and all de jure counties inside it
		# that belong to the defender or their vassals.
		every_in_list = {
			list = target_titles
			custom_tooltip = CONQUEST_CB_TITLE

			# Add targeted duchy title itself if held by defender/vassals
			if = {
				limit = {
					holder ?= {
						OR = {
							this = scope:defender
							target_is_liege_or_above = scope:defender
						}
					}
				}
				add_to_list = titles_taken
			}

			# Add all de jure counties held by defender/vassals
			every_de_jure_county = {
				limit = {
					holder ?= {
						OR = {
							this = scope:defender
							target_is_liege_or_above = scope:defender
						}
					}
				}
				add_to_list = titles_taken
			}
		}

		# Directly usurp all collected titles (counties + duchy), taking baronies.
		every_in_list = {
			list = titles_taken
			change_title_holder = {
				holder = scope:attacker
				change = scope:change
				take_baronies = yes
			}
		}

		resolve_title_and_vassal_change = scope:change

		# Prestige progress for the attacker per target duchy.
		every_in_list = {
			list = target_titles
			scope:attacker = {
				add_prestige_experience = medium_prestige_value
			}
		}

		# Allies' fame.
		modify_allies_of_participants_fame_values = {
			WINNER = scope:attacker
			FAME_BASE = medium_prestige_value
			IS_RELIGIOUS_WAR = no
			WINNER_ALLY_FAME_SCALE = 1
			LOSER_ALLY_FAME_SCALE = 1
		}

		add_truce_attacker_victory_effect = yes
	}

	on_white_peace_desc = {
		desc = basileus_roman_reclamation_cb_white_peace_desc
	}

	on_white_peace = {
		scope:attacker = { show_pow_release_message_effect = yes }

		scope:defender = { accolade_defender_war_end_glory_gain_low_effect = yes }

		scope:attacker = {
			add_prestige = {
				value = minor_prestige_value
				multiply = -1.0
			}
		}

		modify_allies_of_participants_fame_values = {
			WINNER = scope:attacker
			FAME_BASE = medium_prestige_value
			IS_RELIGIOUS_WAR = no
			WINNER_ALLY_FAME_SCALE = 1
			LOSER_ALLY_FAME_SCALE = 1
		}

		add_truce_white_peace_effect = yes
	}

	on_defeat_desc = {
		desc = basileus_roman_reclamation_cb_defeat_desc
	}

	on_defeat = {
		scope:attacker = { show_pow_release_message_effect = yes }

		add_legitimacy_attacker_defeat_effect = yes

		scope:defender = {
			accolade_defender_war_end_glory_gain_med_effect = yes
			add_prestige_war_defender_effect = {
				PRESTIGE_VALUE = major_prestige_value
			}
		}

		scope:attacker = {
			pay_short_term_gold_reparations_effect = { GOLD_VALUE = 5 }
			add_prestige = {
				value = major_prestige_value
				multiply = -1.0
			}
		}

		modify_allies_of_participants_fame_values = {
			WINNER = scope:defender
			FAME_BASE = medium_prestige_value
			IS_RELIGIOUS_WAR = no
			WINNER_ALLY_FAME_SCALE = 1
			LOSER_ALLY_FAME_SCALE = 1
		}

		add_truce_attacker_defeat_effect = yes

		scope:attacker = { save_temporary_scope_as = loser }
		on_lost_aggression_war_discontent_loss = yes
	}

	transfer_behavior = transfer

	on_primary_attacker_death = inherit
	on_primary_defender_death = inherit

	attacker_allies_inherit = yes
	defender_allies_inherit = yes

	war_name      = "BASILEUS_ROMAN_RECLAMATION_WAR_NAME"
	war_name_base = "BASILEUS_ROMAN_RECLAMATION_WAR_NAME_BASE"
	cb_name       = "BASILEUS_ROMAN_RECLAMATION_CB_NAME"

	interface_priority = 110

	use_de_jure_wargoal_only = yes
	attacker_wargoal_percentage = 0.8

	max_defender_score_from_occupation = 150

	ai_score_mult = {
		value = 1
	}
}

# =====================================================================
# BASILEUS — IMPERIAL SUBJUGATION — kingdom-tier casus belli.
# ---------------------------------------------------------------------
# A once-per-lifetime grand war, usable ONLY by the e_byzantium holder while
# the Eastern Reconquest struggle is in its TRIUMPH phase. Targets a kingdom
# (k_) title with de jure land in custom_roman_orient where the defender holds
# at least one county. On victory the attacker seizes the kingdom title (if the
# defender holds it) plus every duchy/county/barony de jure under it held by the
# defender or their vassals.
#
# The once-per-lifetime lock (character flag basileus_imperial_subjugation_used)
# is set only on VICTORY — a white peace or defeat leaves it usable again.
# =====================================================================

basileus_imperial_subjugation_cb = {
	icon = roman_reclamation
	group = conquest

	combine_into_one = yes
	should_show_war_goal_subview = yes
	mutually_exclusive_titles = { always = yes }

	attacker_score_from_occupation_scale = 150
	attacker_score_from_battles_scale = 150
	defender_score_from_battles_scale = 150

	allowed_for_character = {
		has_title = title:e_byzantium
		NOT = { government_has_flag = government_is_landless_adventurer }
		# Once per lifetime (consumed on victory).
		NOT = { has_character_flag = basileus_imperial_subjugation_used }
		# Only during the Triumph phase of the Eastern Reconquest struggle.
		any_character_struggle = {
			is_struggle_type = basileus_eastern_reconquest_struggle
			is_struggle_phase = struggle_eastern_reconquest_phase_triumph
		}
	}

	allowed_against_character = {
		# Any non-Byzantine holder (not the attacker, not under e_byzantium).
		NOR = {
			scope:defender = { this = scope:attacker }
			scope:defender = { target_is_liege_or_above = scope:attacker }
			scope:attacker = { target_is_liege_or_above = scope:defender }
			scope:defender = { has_title = title:e_byzantium }
			scope:defender = { any_liege_or_above = { has_title = title:e_byzantium } }
		}
		# Defender must hold land in the Orient.
		scope:defender = {
			any_realm_county = {
				title_province = { geographical_region = custom_roman_orient }
			}
		}
	}

	target_titles = all
	target_title_tier = kingdom
	target_de_jure_regions_above = yes
	ignore_effect = change_title_holder
	ai_only_against_neighbors = yes

	valid_to_start = {
		scope:target = {
			tier = tier_kingdom
			is_landless_type_title = no
			# Kingdom must have a de jure county in the Orient that the defender
			# (or a vassal) holds.
			any_de_jure_county = {
				title_province = { geographical_region = custom_roman_orient }
				holder ?= {
					OR = {
						this = scope:defender
						target_is_liege_or_above = scope:defender
					}
				}
			}
		}
	}

	should_invalidate = {
		NOT = {
			any_in_list = {
				list = target_titles
				any_in_de_jure_hierarchy = {
					tier = tier_county
					holder = {
						target_is_same_character_or_above = scope:defender
					}
				}
			}
		}
	}

	on_invalidated_desc = msg_invalidate_war_title
	on_invalidated = {}

	cost = {
		prestige = {
			value = major_prestige_value
			multiply = common_cb_prestige_cost_multiplier
		}
	}

	on_declaration = {
		on_declared_war = yes
	}

	# -----------------------------------------------------------------
	# VICTORY — seize the targeted kingdom + everything de jure under it
	# held by the defender or their vassals.
	# -----------------------------------------------------------------
	on_victory_desc = {
		desc = basileus_imperial_subjugation_cb_victory_desc
	}

	on_victory = {
		scope:attacker = { show_pow_release_message_effect = yes }

		add_legitimacy_attacker_victory_effect = yes

		scope:attacker = { accolade_attacker_war_end_glory_gain_med_effect = yes }

		create_title_and_vassal_change = {
			type = conquest
			save_scope_as = change
			add_claim_on_loss = yes
		}

		every_in_list = {
			list = target_titles
			custom_tooltip = CONQUEST_CB_TITLE

			# The kingdom title itself, if held by the defender or a vassal.
			if = {
				limit = {
					holder ?= {
						OR = {
							this = scope:defender
							target_is_liege_or_above = scope:defender
						}
					}
				}
				add_to_list = titles_taken
			}

			# All duchies + counties de jure under it held by defender/vassals.
			every_in_de_jure_hierarchy = {
				limit = {
					OR = { tier = tier_duchy  tier = tier_county }
					holder ?= {
						OR = {
							this = scope:defender
							target_is_liege_or_above = scope:defender
						}
					}
				}
				add_to_list = titles_taken
			}
		}

		every_in_list = {
			list = titles_taken
			change_title_holder = {
				holder = scope:attacker
				change = scope:change
				take_baronies = yes
			}
		}

		resolve_title_and_vassal_change = scope:change

		# Consume the once-per-lifetime use.
		scope:attacker = { add_character_flag = basileus_imperial_subjugation_used }

		scope:attacker = {
			add_prestige_experience = major_prestige_value
		}

		modify_allies_of_participants_fame_values = {
			WINNER = scope:attacker
			FAME_BASE = medium_prestige_value
			IS_RELIGIOUS_WAR = no
			WINNER_ALLY_FAME_SCALE = 1
			LOSER_ALLY_FAME_SCALE = 1
		}

		add_truce_attacker_victory_effect = yes
	}

	on_white_peace_desc = {
		desc = basileus_imperial_subjugation_cb_white_peace_desc
	}

	on_white_peace = {
		scope:attacker = { show_pow_release_message_effect = yes }
		scope:defender = { accolade_defender_war_end_glory_gain_low_effect = yes }
		scope:attacker = {
			add_prestige = {
				value = minor_prestige_value
				multiply = -1.0
			}
		}
		modify_allies_of_participants_fame_values = {
			WINNER = scope:attacker
			FAME_BASE = medium_prestige_value
			IS_RELIGIOUS_WAR = no
			WINNER_ALLY_FAME_SCALE = 1
			LOSER_ALLY_FAME_SCALE = 1
		}
		add_truce_white_peace_effect = yes
	}

	on_defeat_desc = {
		desc = basileus_imperial_subjugation_cb_defeat_desc
	}

	on_defeat = {
		scope:attacker = { show_pow_release_message_effect = yes }
		add_legitimacy_attacker_defeat_effect = yes
		scope:defender = {
			accolade_defender_war_end_glory_gain_med_effect = yes
			add_prestige_war_defender_effect = {
				PRESTIGE_VALUE = major_prestige_value
			}
		}
		scope:attacker = {
			pay_short_term_gold_reparations_effect = { GOLD_VALUE = 5 }
			add_prestige = {
				value = major_prestige_value
				multiply = -1.0
			}
		}
		modify_allies_of_participants_fame_values = {
			WINNER = scope:defender
			FAME_BASE = medium_prestige_value
			IS_RELIGIOUS_WAR = no
			WINNER_ALLY_FAME_SCALE = 1
			LOSER_ALLY_FAME_SCALE = 1
		}
		add_truce_attacker_defeat_effect = yes
		scope:attacker = { save_temporary_scope_as = loser }
		on_lost_aggression_war_discontent_loss = yes
	}

	transfer_behavior = transfer

	on_primary_attacker_death = inherit
	on_primary_defender_death = inherit

	attacker_allies_inherit = yes
	defender_allies_inherit = yes

	war_name      = "BASILEUS_IMPERIAL_SUBJUGATION_WAR_NAME"
	war_name_base = "BASILEUS_IMPERIAL_SUBJUGATION_WAR_NAME_BASE"
	cb_name       = "BASILEUS_IMPERIAL_SUBJUGATION_CB_NAME"

	interface_priority = 115

	max_defender_score_from_occupation = 150

	ai_score_mult = {
		value = 1
	}
}

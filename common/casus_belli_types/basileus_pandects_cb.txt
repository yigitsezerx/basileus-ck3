# Basileus Pandects CB
pandects_betrayal_cb = {
	icon = dissolution_war
	group = civil_war
	target_titles = claim
	allow_hostages = yes
	target_top_liege_if_outside_realm = no
	
	# Allow allies to join
	attacker_allies_inherit = yes
	defender_allies_inherit = yes

	# This is a scripted CB, so we keep allowed checks hidden from UI
	allowed_for_character = {
		always = no
	}

	allowed_against_character = {
		always = no
	}

	on_victory_desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					scope:defender = { 
						government_has_flag = government_is_mandala
						is_local_player = yes 
					}
				}
				desc = mandala_depose_war_victory_desc_defender
			}
			triggered_desc = {
				trigger = {
					scope:defender = { 
						government_has_flag = government_is_mandala
						is_local_player = no 
					}
				}
				desc = mandala_depose_war_victory_desc
			}
			triggered_desc = {
				trigger = {
					scope:defender = { is_local_player = yes }
					OR = {
						NOT = { exists = scope:attacker.var:initial_target }
						scope:defender = scope:attacker.var:initial_target
						scope:defender.primary_title.tier <= scope:attacker.var:target_tier
					}
				}
				desc = depose_war_victory_desc_defender
			}
			triggered_desc = {
				trigger = {
					OR = {
						NOT = { exists = scope:attacker.var:initial_target }
						scope:defender = scope:attacker.var:initial_target
						scope:defender.primary_title.tier <= scope:attacker.var:target_tier
					}
				}
				desc = depose_war_victory_desc
			}
		}
	}	

	# On Victory (Attacker wins)
	on_victory = {
		scope:attacker = { show_pow_release_message_effect = yes }
		
		# Standard title transfer for claim war
		create_title_and_vassal_change = {
			type = conquest_claim
			save_scope_as = change
			add_claim_on_loss = yes
		}
		
		setup_claim_cb = {
			titles = target_titles
			attacker = scope:attacker
			defender = scope:defender
			claimant = scope:claimant
			change = scope:change
		}

		resolve_title_and_vassal_change = scope:change
		
		# Truce
		add_truce_attacker_victory_effect = yes
	}

	on_defeat_desc = {
		first_valid = {
			triggered_desc = {
				trigger = { scope:defender = { is_local_player = yes } }
				desc = depose_war_defeat_desc_defender
			}
			desc = depose_war_defeat_desc
		}

	}	

	# On Defeat (Attacker loses, Defender wins)
	on_defeat = {
		scope:attacker = { show_pow_release_message_effect = yes }
		
		# Imprison all rebel attackers (leader and co-belligerents/allies)
		every_war_attacker = {
			limit = {
				is_alive = yes
				exists = liege
				any_liege_or_above = { this = scope:defender }
			}
			scope:defender = {
				imprison_character_effect = {
					TARGET = prev
				}
			}
		}
		scope:defender = {
			imprison_character_effect = {
				TARGET = scope:attacker
			}
		}
		
		# Imprison all characters in the detractors list of the Renovatio story cycle
		random_owned_story = {
			limit = { type = story_renovatio }
			save_scope_as = current_renovatio_story
		}
		if = {
			limit = {
				exists = scope:current_renovatio_story
				scope:current_renovatio_story = {
					has_variable_list = detractors
				}
			}
			scope:current_renovatio_story = {
				every_in_list = {
					variable = detractors
					limit = {
						is_alive = yes
					}
					save_scope_as = detractor_to_imprison
					scope:defender = {
						imprison_character_effect = {
							TARGET = scope:detractor_to_imprison
						}
					}
				}
			}
		}
		
		# Truce
		add_truce_attacker_defeat_effect = yes
		
		# Trigger the Finale Event for the Emperor
		if = {
			limit = {
				scope:defender = { has_title = title:e_byzantium }
			}
			scope:defender = {
				trigger_event = {
					id = pandects_renovatio.0007
					days = { 1 2 }
				}
			}
		}
	}

	# On White Peace (Emperor failed to crush the rebellion)
	on_white_peace = {
		scope:attacker = { show_pow_release_message_effect = yes }
		add_truce_white_peace_effect = yes
		
		# White Peace counts as a victory for detractors
		if = {
			limit = {
				scope:defender = { has_title = title:e_byzantium }
			}
			scope:defender = {
				# Fail the story
				trigger_event = {
					id = pandects_renovatio.0006 #?
					# Actually, maybe a custom failure event or just Option A effects
				}
				# For now, let's just apply the Option A failure effects manually or trigger a new event
				custom_tooltip = PANDECTS_WHITE_PEACE_FAILURE_TT
				
				# Revert language
				title:e_byzantium = {
					holder = {
						# No set_court_language here to avoid loc issues, or use a better way
						custom_tooltip = PANDECTS_REVERT_LANGUAGE_TT
					}
				}
				
				# End Story
				random_owned_story = {
					limit = { type = story_renovatio }
					end_story = yes
				}
				
				add_character_modifier = {
					modifier = basileus_broken_spirit
					years = 5
				}
			}
		}
	}

	transfer_behavior = transfer
	on_primary_attacker_death = inherit
	on_primary_defender_death = inherit

	war_name = "PANDECTS_BETRAYAL_WAR_NAME"
	interface_priority = 100
}
